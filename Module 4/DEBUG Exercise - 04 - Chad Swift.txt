// This pseudocode segment is intended to compute and display
// the cost of home ownership for any number of users
// The program ends when a user enters 0 for the mortgage payment
START
   Declarations
      num mortgagePayment
      num utilities
      num taxes
      num upkeep
      num total
   mortgagePayment = startUp(mortgagePayment)
   WHILE mortgagePayment <> 0
      MainLoop(mortgagePayment, utilities, taxes, upkeep, total)
      mortgagePayment = startUp(mortgagePayment)
   ENDWHILE
   finishUp()
STOP

startUp()
   OUTPUT "Enter your mortgage payment or 0 to quit"
   INPUT mortgagePayment
RETURN mortgagePayment

mainLoop()
   OUTPUT "Enter utilities"
   INPUT utilities
   OUTPUT "Enter taxes"
   INPUT taxes
   OUTPUT "Enter amount for upkeep"
   INPUT upkeep
   total = mortgagePayment + utilities + taxes + upkeep
   OUTPUT "Utilities are $" + utilities + ". Taxes are $" + taxes + ". Upkeep is $" + upkeep + ". Total is " + total
RETURN

finishUp()
   OUTPUT "End of program"
RETURN
