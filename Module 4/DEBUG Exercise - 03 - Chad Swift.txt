// This pseudocode is intended to determine whether students have
// passed or failed a course; student needs to average 60 or
// more on two tests. 
START
   Declarations
      string name
      num firstTest
      num secondTest
      num average = (firstTest + secondTest) / 2
      num passingGrade = 60
   OUTPUT "Enter student name:"
   INPUT name
   OUTPUT "Enter first score or 0 to quit for " + name
   INPUT firstTest
   WHILE firstTest <> 0
      OUTPUT "Enter second score for " + name
      INPUT secondTest
      OUTPUT "First test score was " + firstTest + ". Second test score was " + secondTest + ". Average for " + name + " is " + average
      OUTPUT passingGrade + " is passing"
      IF average >= passingGrade
         OUTPUT name + "has passsed"
      ELSE
         OUTPUT name + "has failed"
      ENDIF
      OUTPUT "Enter next student name"
      INPUT name
      OUTPUT "Enter first score or 0 to quit for " + name
      INPUT firstTest
   ENDWHILE
STOP
